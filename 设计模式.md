### 单例模式

##### 饿汉模式
直接生成一个实例
```
var userInfo =  (function (){
			function x (name,sex,age){
				this.name = name;
				this.sex = sex;
				this.Num = function (n){
					var num = n*n;
					console.log(num);
				}
			}
			// return new x('王先森','男','24');
			var privateX = new x('王先森','男','24');
			publicX = privateX;
		})()
		// document.write('用户名称：'+userInfo.name);
		// userInfo.Num(12);
		console.log('用户名称：'+publicX.name);
		publicX.Num(12);
```
##### 懒汉模式
需要时再生成实例
```
var userInfo =  (function (){
			function x (name,sex,age){
				this.name = name;
				this.sex = sex;
				this.age = age;
				this.Num = function (n){
					var num = n*n;
					console.log(num);
				}
			}
			if (typeof publicX == "undefined" ) {
				var privateX = new x('王先森','男','24');
				publicX = privateX;
			}else{
				console.log('实例已经创建过')
			}
		})()
		console.log('用户名称：'+publicX.name);
		publicX.Num(10);
```
